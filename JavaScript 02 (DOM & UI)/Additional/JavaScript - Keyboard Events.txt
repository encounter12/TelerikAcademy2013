self.searchKeyPress = function searchKeyPress(e) {
    if (typeof e === "undefined" && window.event) {
        e = window.event;
    }
    if (e.keyCode === 13) {
        document.getElementById("process").click();
        return false;
    }
    if (e.keyCode === 27) {
        document.getElementById("clear").click();
        return false;
    }
    }
